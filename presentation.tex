\section{Future work}
\begin{frame}{Future work}

\begin{block}{Article2vec}
\begin{itemize}
\item Pretrained embeddings across markets?
\item Continue exploring visualizations.

\end{itemize}

\begin{block}{Predicting browsing \& Reco}
\begin{itemize}

    \item Remove the user embedding?
\end{itemize}
\end{block}
